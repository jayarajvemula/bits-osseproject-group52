function Get-AreaofTriangle
{
      Param
    (
         [Parameter(Mandatory=$true, Position=0)]   
         [ValidateRange(0,10000)]
         [int] $Height,
         [Parameter(Mandatory=$true, Position=1)]        
         [ValidateRange(0,10000)]
         [int] $Base
    )

    
    Write-Host "Area of Triangle is : "  ($Height * $Base)/2

}

Get-AreaofTriangle 10 10
